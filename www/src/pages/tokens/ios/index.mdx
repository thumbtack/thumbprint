import { version, homepage } from '../../../../../packages/thumbprint-tokens/package.json';
import {
    formatId,
    formatValue,
    formatSectionName,
} from '../../../../../packages/thumbprint-tokens/src/helpers/ios.js';
import classNames from 'classnames';
import { camelCase } from 'lodash';
import { graphql } from 'gatsby';
import PackageTable from 'components/package-table';
import TabNav from 'components/thumbprint-tokens/tab-nav';
import TokenSection from 'components/thumbprint-tokens/token-section';

export const pageQuery = graphql`
    {
        allThumbprintToken(
            filter: { tokens: { elemMatch: { value: { ios: { ne: null } } } } }
            sort: { fields: name }
        ) {
            nodes {
                name
                description
                tokens {
                    id
                    type
                    deprecated
                    description {
                        ios
                    }
                    value {
                        ios
                        web
                    }
                    group
                }
            }
        }
    }
`;

<TabNav />

<PackageTable
    version={version}
    packageName="ThumbprintTokens"
    sourceDirectory={homepage}
    platform="ios"
    importStatement="import ThumbprintTokens"
/>

<div>
    {props.data.allThumbprintToken.nodes.map(section => (
        <TokenSection
            section={section}
            formatId={id => `${formatSectionName(section.name)}.${formatId(id)}`}
            formatValue={value => formatValue(value).replace('UIColor = ', '')}
            key={section.name}
            platform="ios"
        />
    ))}
</div>
