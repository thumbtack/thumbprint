## Link styles

The `Link` component can be used to render plain text links.

```jsx
<React.Fragment>
    Learn more about{' '}
    <FutureLink href="https://www.facebook.com/Thumbtack/">Thumbtack on Facebook</FutureLink>.
</React.Fragment>
```

export const pageQuery = graphql`
    {
        # Get links to by path to display in the navbar.
        platformNav: allSitePage(filter: { path: { glob: "/components/link/*/" } }) {
            edges {
                node {
                    ...PlatformNavFragment
                }
            }
        }
        # Get package information by NPM package name.
        packageTable: thumbprintComponent(name: { eq: "@thumbtack/thumbprint-react" }) {
            ...PackageTableFragment
        }
        # Get component props by path to component file.
        reactComponentProps: allFile(
            filter: { relativePath: { in: ["thumbprint-react/components/Link/index.tsx"] } }
        ) {
            ...ReactComponentPropsFragment
        }
    }
`;
